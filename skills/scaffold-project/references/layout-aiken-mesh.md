# Layout: Aiken + Mesh SDK (Stack 2)

Stack 2 targets a TypeScript team that wants the broadest tutorial coverage and a batteries-included off-chain toolchain. Aiken on-chain emits a CIP-57 `plutus.json` blueprint; Mesh SDK off-chain consumes it. Mesh ships React components for wallet connect and relies on a WASM core for serialization.

### Hello-world validator skeleton -- `onchain/validators/hello.ak`

A trivial validator that succeeds when the spending transaction is signed by a specific public-key hash stored in the datum. Enough to exercise the full on-chain / off-chain loop without distracting business logic.

```aiken
// onchain/validators/hello.ak
//
// Hello-world validator: spend allowed only if the signer in the datum has
// signed the transaction. This is the simplest non-trivial validator -- it
// exists to prove the build + deploy + spend cycle works end-to-end. Replace
// with real business logic by handing off to the write-validator skill.

use aiken/transaction.{Transaction, ScriptContext, Spend}
use aiken/transaction/credential.{VerificationKeyHash}
use aiken/list

pub type HelloDatum {
  owner: VerificationKeyHash,
}

validator hello {
  spend(
    datum: Option<HelloDatum>,
    _redeemer: Data,
    _own_ref: Data,
    tx: Transaction,
  ) -> Bool {
    expect Some(HelloDatum { owner }) = datum
    list.has(tx.extra_signatories, owner)
  }
}

// Unit tests live alongside the validator.
test owner_can_spend() {
  let owner = #"00112233445566778899aabbccddeeff00112233445566778899aabb"
  let datum = HelloDatum { owner }
  // ... build a mock Transaction with `owner` in extra_signatories and assert.
  // See https://aiken-lang.org/language-tour/testing for the mocking helpers,
  // or ${CLAUDE_SKILL_DIR}/../../docs/sources/aiken/ for the testing chapter.
  True
}
```

Build with `aiken build` from the `onchain/` directory. The resulting `plutus.json` is the contract with the off-chain side.

### Hello-world off-chain script -- `offchain/src/blueprint.ts`

```typescript
// offchain/src/blueprint.ts
//
// Loads the CIP-57 plutus.json blueprint emitted by `aiken build` and exposes
// the compiled validator CBOR plus a typed wrapper. Every other off-chain
// module imports the validator from here -- never re-derive the script hash.

import { readFileSync } from "node:fs";
import { fileURLToPath } from "node:url";
import { dirname, resolve } from "node:path";

const here = dirname(fileURLToPath(import.meta.url));
const blueprintPath = resolve(here, "../../onchain/plutus.json");
//                                ^^^^^^^^^^^^^^^^^^^^^^^^^^
// Monorepo path. For single-repo, change to "../plutus.json".

type Blueprint = {
  validators: Array<{
    title: string;
    compiledCode: string;       // hex-encoded CBOR
    hash: string;               // script hash
  }>;
};

const blueprint = JSON.parse(readFileSync(blueprintPath, "utf-8")) as Blueprint;

export function getValidator(title: string) {
  const v = blueprint.validators.find((x) => x.title === title);
  if (!v) {
    throw new Error(
      `Validator '${title}' not found in plutus.json. Did you run 'aiken build'?`,
    );
  }
  return { compiledCode: v.compiledCode, hash: v.hash };
}
```

### Hello-world off-chain script -- `offchain/src/provider.ts`

```typescript
// offchain/src/provider.ts
//
// Switches between Yaci DevKit (local devnet) and Blockfrost (preview /
// preprod / mainnet) based on CARDANO_NETWORK. Devnet is the default; mainnet
// requires explicit opt-in via the env var.

import "dotenv/config";
import { BlockfrostProvider } from "@meshsdk/core";

export function getProvider() {
  const network = process.env.CARDANO_NETWORK ?? "devnet";

  if (network === "devnet") {
    // Yaci Store exposes a Blockfrost-compatible API. The base URL points at
    // the local devnet; no project ID required.
    const url = process.env.YACI_STORE_URL ?? "http://localhost:10000";
    return new BlockfrostProvider(url + "/api/v1/");
  }

  const projectId = process.env.BLOCKFROST_PROJECT_ID;
  if (!projectId) {
    throw new Error("BLOCKFROST_PROJECT_ID required when CARDANO_NETWORK is not devnet");
  }
  return new BlockfrostProvider(projectId);
}
```

### Hello-world off-chain script -- `offchain/src/hello/lock.ts`

```typescript
// offchain/src/hello/lock.ts
//
// Locks 5 ADA at the hello validator with a datum containing the dev wallet's
// pubkey hash. Run with: npm run tx:lock

import "dotenv/config";
import {
  MeshTxBuilder,
  MeshWallet,
  resolvePaymentKeyHash,
  resolvePlutusScriptAddress,
  serializePlutusScript,
} from "@meshsdk/core";
import { getProvider } from "../provider.js";
import { getValidator } from "../blueprint.js";

async function main() {
  const provider = getProvider();
  const wallet = new MeshWallet({
    networkId: 0,                        // 0 = testnet/devnet, 1 = mainnet
    fetcher: provider,
    submitter: provider,
    key: { type: "mnemonic", words: process.env.DEV_WALLET_MNEMONIC!.split(" ") },
  });

  const hello = getValidator("hello.hello.spend");      // title in plutus.json
  const scriptAddr = serializePlutusScript({ code: hello.compiledCode, version: "V3" }).address;

  const senderAddr = await wallet.getChangeAddress();
  const ownerHash = resolvePaymentKeyHash(senderAddr);

  const utxos = await wallet.getUtxos();

  const txBuilder = new MeshTxBuilder({ fetcher: provider, submitter: provider });
  const unsignedTx = await txBuilder
    .txOut(scriptAddr, [{ unit: "lovelace", quantity: "5000000" }])
    .txOutDatumHashValue({ owner: ownerHash })       // simplified; see Mesh docs for inline datum
    .changeAddress(senderAddr)
    .selectUtxosFrom(utxos)
    .complete();

  const signedTx = await wallet.signTx(unsignedTx);
  const txHash = await wallet.submitTx(signedTx);
  console.log("Locked. Tx hash:", txHash);
}

main().catch((err) => {
  console.error(err);
  process.exit(1);
});
```

A matching `redeem.ts` queries the script address for UTxOs and spends the locked one back to the wallet, providing the datum and an empty redeemer. The body follows the standard Mesh spend pattern; the developer fills it in by handing off to `build-transaction`.

## Network config patterns

Three places to flip when switching networks:

1. `CARDANO_NETWORK` in `.env` (devnet | preview | preprod | mainnet). Default to a testnet; never mainnet.
2. The provider factory in `provider.ts` (already shown above) switches the Blockfrost URL based on the env var. On devnet the URL points at Yaci Store; on preview/preprod/mainnet, at the matching Blockfrost endpoint.
3. `MeshWallet.networkId`: 0 for any testnet (devnet, preview, preprod); 1 for mainnet.

Each non-devnet network needs its own Blockfrost project ID from https://blockfrost.io (each has a distinct prefix: `preview...`, `preprod...`, `mainnet...`). Faucets for preview and preprod live at https://docs.cardano.org/cardano-testnets/tools/faucet вЂ” pick the matching selector on the page. Yaci DevKit has its own built-in faucet via `yaci-cli faucet send`.

## Frontend (optional, default ON in the scaffold)

When the developer opts in, scaffold a Next.js App Router app under `frontend/` (same monorepo as off-chain). Mesh's `<CardanoWallet />` React component handles wallet discovery and connection out of the box; tx building uses the same `MeshTxBuilder` pattern as the backend code.

CIP-30 details (beyond `<CardanoWallet />`): hand off to `connect-wallet`. Keep `BLOCKFROST_PROJECT_ID` out of `NEXT_PUBLIC_*` env. Front Blockfrost calls behind a Next.js route handler so the project ID stays server-side. Wallet calls remain client-side because the wallet has no server.
